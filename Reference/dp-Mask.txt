struct city{
	int a, b, c;
};
vector<city>arr;
int dis(city A, city B){
	return abs(A.a - B.a) + abs(A.b - B.b) + max(0, B.c - A.c);
}
int n, dp[18][(1 << 18)];
int solve(int lst, int mask){
	if (mask == (1 << n) - 1)
		return dis(arr[lst], arr[0]);
	int &ret = dp[lst][mask];
	if (~ret)return ret; ret = oo;
	for (int i = 0; i < n; i++){
		ret = min(ret, solve(i, mask | (1 << i)) + dis(arr[lst], arr[i]));
	}
	return ret;
}
int main(){
