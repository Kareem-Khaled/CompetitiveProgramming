// Know first num greater than you 

int main(){
	ll n; cin >> n;
	vector<ll>v(n + 1), ans(n + 1);
	for (int i = 1; i <= n; i++)cin >> v[i];
	stack<ll>st; st.push(1); ans[1] = -1;
	for (int i = 2; i <= n; i++){
		while (sz(st) && v[i] > v[st.top()]){
			ans[st.top()] = i;
			st.pop();
		}
		st.push(i);
		ans[i] = -1;
	}
	int q; cin >> q;
	while (q--){
		int x;cin >> x;
		cout << ans[x] << endl;
	}
}
