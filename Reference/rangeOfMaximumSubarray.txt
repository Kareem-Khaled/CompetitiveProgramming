void findMaxSubArrayIndex(){
	int l = 0, r = 0, curmx = 0, premx = arr[0], l2 = 0, mx = 0;
	for (int i = 0; i < n; i++){
		curmx += arr[i];
		if (curmx < 0){
			l = i + 1; curmx = 0;
		}
		else if (curmx == mx && i - l > r - l2){
			l2 = l;	r = i;
		}
		else if (curmx > premx){
			r = i;
			l2 = l;
			premx = curmx;
		}
		mx = max(mx, curmx);
	}
}
